function generateIKTETitle(isu, bidang, tahap) { const b = bidang.toLowerCase(); const i = isu.toLowerCase(); let template = ""; // Domain Mapping Logic if (b.includes('akademik') || b.includes('sains') || b.includes('matematik') || b.includes('bahasa') || b.includes('sejarah') || b.includes('pendidikan islam')) { template = `Meningkatkan Penguasaan ${isu} melalui Teknik Inovatif dalam kalangan Murid ${tahap}`; } else if (b.includes('kaunseling') || b.includes('psikososial')) { template = `Meningkatkan ${i.includes('keyakinan') || i.includes('motivasi') ? 'Keyakinan Diri' : 'Kesejahteraan Emosi'} melalui Modul Bimbingan dalam kalangan Murid ${tahap}`; } else if (b.includes('disiplin') || b.includes('ponteng')) { template = `Mengurangkan Tingkah Laku ${isu} melalui Pendekatan Mentor-Mentee dalam kalangan Murid ${tahap}`; } else if (b.includes('hem') || b.includes('kehadiran')) { template = `Meningkatkan ${isu} melalui Program Intervensi Strategik dalam kalangan Murid ${tahap}`; } else if (b.includes('kokurikulum') || b.includes('sukan') || b.includes('unit beruniform')) { template = `Meningkatkan Penyertaan ${isu} melalui Sistem Merit dalam kalangan Murid ${tahap}`; } else if (b.includes('kebersihan') || b.includes('keceriaan')) { template = `Membudayakan Amalan ${isu} melalui Program Kelas Ceria dalam kalangan Murid ${tahap}`; } else if (b.includes('nilam') || b.includes('pusat sumber')) { template = `Meningkatkan Minat ${isu} melalui Program NILAM Digital dalam kalangan Murid ${tahap}`; } else if (b.includes('mbpk') || b.includes('pendidikan khas')) { template = `Meningkatkan Kemahiran ${isu} melalui Modul Pembelajaran Berasaskan Bahan Bantu Belajar dalam kalangan Murid ${tahap}`; } else { template = `Meningkatkan ${isu} melalui Strategi PdP Berfokus dalam kalangan Murid ${tahap}`; } return template; }
